Curso de Git y Github en platzi

Comandos para Git

    git init 
// Inicializar tu repositorio local
        
    git add
// Agregar cambios a stage de git, siempre se tiene que hacer git add al archivo antes del commit
        
    git commit -m
// Mandar todos tus cambios al repositorio local
        
    git push
// Mandar tu repositorio local a repositorio remoto
        
    git log
// checar los commits que se han hecho a el archivo o la rama completa
        
    git fetch
// jalar cambios de repositorio remoto a local, pero es mejor usar pull
        
    git pull origin master
// Tener la ultima version del repositorio remoto
        
    git push origin master
// Empujar a origin (remote) la rama master
        
    git merge
// combinar dos ramas distintas
        
    git remote -v
// conocer si hay repositorio remoto
        
    git branch
// ver las ramas que existen
        
    git switch -c "nombre de nueva branch"
// crear una rama nueva y brincar a ella
        
    git checkout
// cambiar entre ramas
        
    git tag, git tag -d nombre-del-tag
//Te muestra los tags creados  // borra tags creados
        
    git tag -a v0.1 -m "Version sin libreria javascript" db44ce9 
//Crea un tag (una version de un commit) le pone un nombre (v0,1) y un mensaje y el numero del commit
        
     git show-branch --all 
//muestra ramas y su historia
